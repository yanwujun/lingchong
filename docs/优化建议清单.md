def smooth_move(self, target_pos):

from PyQt5.QtMultimedia import QSoundEffect
from PyQt5.QtCore import QUrl

class SoundManager:
    def __init__(self):
        self.sounds = {}
        self.load_sounds()
    
    def load_sounds(self):
        """加载音效"""
        sound_files = {
            'click': 'assets/sounds/click.wav',
            'alert': 'assets/sounds/alert.wav',
            'complete': 'assets/sounds/complete.wav'

                self.sounds[name] = sound
    
    def play(self, sound_name):
        """播放音效"""
        if sound_name in self.sounds:
            self.sounds[sound_name].play()

### 7. 提醒间隔固定 🟡
**当前问题**：
- 提醒检查间隔固定60秒
- 可能错过精确时间的提醒

**优化方案**：
```python
# 智能计算下次检查时间
def calculate_next_check_interval(self):
    """计算下次检查间隔"""
    # 获取最近的提醒时间
    next_reminder = self.database.get_next_reminder()
    
    if not next_reminder:
        return 60000  # 默认60秒
    
    # 计算到下次提醒的时间
    now = datetime.now()
    remind_time = parse_datetime(next_reminder['remind_time'])
    delta = (remind_time - now).total_seconds()
    
    if delta <= 0:
        return 1000  # 1秒，立即检查
    elif delta <= 60:
        return int(delta * 1000)  # 精确到秒
    else:
        return 60000  # 60秒
```

**优化位置**：
- `src/reminder.py` - 动态调整检查间隔

def mouseMoveEvent(self, event):
    """鼠标移动事件"""
    if self.is_dragging and event.buttons() == Qt.LeftButton:
        new_pos = event.globalPos() - self.drag_position
        
        # 限制在屏幕范围内
        screen = self.screen().geometry()
        x = max(0, min(new_pos.x(), screen.width() - self.width()))
        y = max(0, min(new_pos.y(), screen.height() - self.height()))
        
        self.move(x, y)
        event.accept()

# 使用上下文管理器
from contextlib import contextmanager

@contextmanager
def get_connection(self):
    """获取数据库连接的上下文管理器"""
    self.connect()
    try:
        yield self.conn
    finally:
        self.conn.commit()
